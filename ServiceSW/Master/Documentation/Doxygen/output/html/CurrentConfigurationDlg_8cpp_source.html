00152                                                    Global::tTranslatableStringList() << m_SubModuleName.toString());
00153         mp_SubModule = const_cast<ServiceDataManager::CModule*>(mp_Module)->GetSubModuleInfo(m_SubModuleName.toString());
00154         int Count = mp_SubModule->GetNumberOfParameters();
00155         ServiceDataManager::Parameter_t *Param;
00156         mp_ParameterDlg->ClearModel();
00157 
00158         for(int i=0; i<Count; i++) {
00159             Param = mp_SubModule->GetParameterInfo(i);
00160             mp_ParameterDlg->InitDialog(Param);
00161         }
00162 
00163         mp_ParameterDlg->SetDialogTitle(mp_SubModule->GetSubModuleName());
00164         mp_ParameterDlg->resize(380, 420);
00165         mp_ParameterDlg->setModal(true);
