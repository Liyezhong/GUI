00149             if (SubModuleTwo && *SubModuleOne == *SubModuleTwo) {
00150                 SameSubModule<<SubModuleOne->GetSubModuleName();

00152         }
00153

00155             (void)ModuleOne.DeleteSubModule(SameSubModule.at(k));
00156             (void)ModuleTwo.DeleteSubModule(SameSubModule.at(k));
00157         }
00158 
00159         if (ModuleOne.GetNumberofSubModules() == 0) {
00160             if (ModuleOne.GetDateOfProduction() == ModuleTwo.GetDateOfProduction()
00161                     && ModuleOne.GetSerialNumber() == ModuleTwo.GetSerialNumber()
00162                     && ModuleOne.GetOperatingHours() == ModuleTwo.GetOperatingHours()) {
00163                 continue;

00165         }
00166 
00167         (void) ModuleListClone1.AddModule(&ModuleOne);
00168         (void) ModuleListClone2.AddModule(&ModuleTwo);
